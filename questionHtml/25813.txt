    &lt;name&gt;hadoop.tmp.dir&lt;/name&gt;

    &lt;/configuration&gt;</code></p>

    &lt;/property&gt;
    &lt;/configuration&gt;</code></p>

    <code>&lt;configuration&gt;
    &lt;property&gt;
    &lt;name&gt;dfs.replication&lt;/name&gt;

    &lt;property&gt;

    &lt;/property&gt;
    &lt;/configuration&gt;</code></p>
