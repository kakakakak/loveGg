        &lt;value&gt;shisoft-hadoopmaster:8088&lt;/value&gt;
    &lt;/property&gt;
    &lt;property&gt;
        &lt;name&gt;yarn.log-aggregation-enable&lt;/name&gt;
        &lt;value&gt;true&lt;/value&gt;
    &lt;/property&gt;
&lt;/configuration&gt;   
</code></pre>

<p>I wonder what is it trying to connect and get refused. And how to fix it.</p>

<p>Thank you.</p>

<p><strong>UPDATE</strong></p>

<p>I found this issue was because Application Manager was bind to wrong address <code>yarn.resourcemanager.webapp.address</code>. But I have already set this in the yarn-core.xml
The way to fix it is to set this value before send this job like</p>

<pre><code>final Configuration conf = new Configuration();
conf.set("fs.defaultFS", "hdfs://shisoft-hadoopmaster:49000/");
conf.set("mapreduce.framework.name", "yarn");
conf.set("yarn.resourcemanager.address", "shisoft-hadoopmaster:8032");
conf.set("yarn.resourcemanager.scheduler.address", "shisoft-hadoopmaster:8030");
conf.set("yarn.resourcemanager.webapp.address", "shisoft-hadoopmaster:8088");
conf.set("mapreduce.jobhistory.webapp.address", "shisoft-hadoopmaster:19888");
...
 final Job job = new Job(conf, "user news word count");
...
</code></pre>

<p>I don't know why I have to do this since it have already been set in node configuration files.</p>
