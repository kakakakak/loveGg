                    s++;
                }

                else{

                    s++;
                    }

            }
            else if(opt_type.get(s).equalsIgnoreCase("Double"))
                    {


 if(args.get(i).getClass().getSimpleName().toString().equalsIgnoreCase("Integer"))
                     {
                    opt_params.add((Double)((Integer)args.get(i)+0.0));
                    s++;
                     } 
                  else{throw new IllegalArgumentException("Opt arguments do   
 not match for fn:"+func_name+", pls enter opt arguments in right order"); }
                    }
            else{throw new IllegalArgumentException("Opt arguments do not match 
 for fn:"+func_name+", pls enter opt arguments in right order");}

        } 


        List&lt;Object&gt; ta_argl = new ArrayList&lt;Object&gt;();
        ta_argl.add(new Integer(0));
        ta_argl.add(new Integer((int)input.size() - 1));
        for(double[]in: ip_list)
        {
         ta_argl.add(in);
        }

        if(opt_noofparams!=0)
        {ta_argl.addAll(opt_params);}
        ta_argl.add(outStart);
        ta_argl.add(outLen);

        for (Map.Entry&lt;String, Object&gt; entry : outputParams.entrySet()) 
            {
             ta_argl.add(entry.getValue());
            }


            RetCode rc = RetCode.Success;
        try {
            rc = (RetCode)func_ref.invoke(core, ta_argl.toArray());
            } catch (Exception e)

            assert false : "I died in ta-lib, but Java made me a zombie...";
            }

        assert rc == RetCode.Success : "ret code from " + func_name;



        if (outLen.value == 0) return null;

        outputParams.clear();
        ip_list.clear();
        opt_params.clear();
        opt_type.clear();
        ip_colmns.clear();
        Runtime.getRuntime().gc();
        return ret;

    }

<p>Probably a problem with the BZip codec - the API does note that it's rather memory hungry:</p>

<p><ul>
<li><a href="http://hadoop.apache.org/common/docs/r0.20.0/api/org/apache/hadoop/io/compress/bzip2/CBZip2OutputStream.html" rel="nofollow">http://hadoop.apache.org/common/docs/r0.20.0/api/org/apache/hadoop/io/compress/bzip2/CBZip2OutputStream.html</a></p>

<blockquote>
  <p><em>The compression requires large amounts of memory</em></li>
  </ul>
  When you increased the memory with <code>-Xms2048m</code> did you set the options for the pig grunt shell, or for the map/reduce jobs?</p>
</blockquote>

<pre><code>set mapred.child.java.opts=-Xmx2048m
</code></pre>

<p>You can check by looking in the JobTracker, find the job that failed, open the job.xml and locate the value of <code>mapred.child.java.opts</code></p>
